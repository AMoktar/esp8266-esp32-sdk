118 bool SinricProThermostat::handleRequest(const char* deviceId, const char* action, JsonObject &request_value, JsonObject &response_value) {
119  if (strcmp(deviceId, this->deviceId) != 0) return false;
120  if (SinricProDevice::handleRequest(deviceId, action, request_value, response_value)) return true;
121 
122  bool success = false;
123  String actionString = String(action);
124 
125  if (actionString == "targetTemperature" && targetTemperatureCallback) {
126  float temperature;
127  if (request_value.containsKey("temperature")) {
128  temperature = request_value["temperature"];
129  } else {
130  temperature = 1;
131  }
132  success = targetTemperatureCallback(String(deviceId), temperature);
133  response_value["temperature"] = temperature;
134  return success;
135  }
136 
137  if (actionString == "adjustTargetTemperature" && adjustTargetTemperatureCallback) {
138  float temperatureDelta = request_value["temperature"];
139  success = adjustTargetTemperatureCallback(String(deviceId), temperatureDelta);
140  response_value["temperature"] = temperatureDelta;
141  return success;
142  }
143 
144  if (actionString == "setThermostatMode" && thermostatModeCallback) {
145  String thermostatMode = request_value["thermostatMode"] | "";
146  success = thermostatModeCallback(String(deviceId), thermostatMode);
147  response_value["thermostatMode"] = thermostatMode;
148  return success;
149  }
150 
151  return success;
152 }
