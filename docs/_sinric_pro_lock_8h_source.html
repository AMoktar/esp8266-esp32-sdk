33 bool SinricProLock::handleRequest(const char* deviceId, const char* action, JsonObject &request_value, JsonObject &response_value) {
34  if (strcmp(deviceId, this->deviceId) != 0) return false;
35  bool success = false;
36  String actionString = String(action);
37 
38  if (actionString == "setLockState" && lockStateCallback) {
39  bool lockState = request_value["state"]=="lock"?true:false;
40  success = lockStateCallback(String(deviceId), lockState);
41  response_value["state"] = success?lockState?"LOCKED":"UNLOCKED":"JAMMED";
42  return success;
43  }
44  return success;
45 }
